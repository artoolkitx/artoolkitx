31typedef enum nx_json_type {
32 NX_JSON_NULL, // this is null value
33 NX_JSON_OBJECT, // this is an object; properties can be found in child nodes
34 NX_JSON_ARRAY, // this is an array; items can be found in child nodes
35 NX_JSON_STRING, // this is a string; value can be found in text_value field
36 NX_JSON_INTEGER, // this is an integer; value can be found in int_value field
37 NX_JSON_DOUBLE, // this is a double; value can be found in dbl_value field
38 NX_JSON_BOOL // this is a boolean; value can be found in int_value field
39} nx_json_type;
40
41typedef struct nx_json {
42 nx_json_type type; // type of json node, see above
43 const char* key; // key of the property; for object's children only
44 const char* text_value; // text value of STRING node
45 long int_value; // the value of INTEGER or BOOL node
46 double dbl_value; // the value of DOUBLE node
47 int length; // number of children of OBJECT or ARRAY
48 struct nx_json* child; // points to first child
49 struct nx_json* next; // points to next child
50 struct nx_json* last_child;
51} nx_json;
